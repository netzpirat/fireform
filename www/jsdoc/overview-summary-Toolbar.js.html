/**
 * the fireform toolbar.
 *
 * @constructor
 */
function Toolbar() {

   // load preferences
   this.preferences = new Preferences();
   
   // observe user change
   var prefChangeObserver = {
      observe :
      function(aSubject, aTopic, aData) {
         toolbar.createUserMenu();
      }
   };
   
   // add observer to preference
   var query = Components.classes["@mozilla.org/preferences-service;1"].getService(Components.interfaces.nsIPrefService).getBranch(null).QueryInterface(Components.interfaces["nsIPrefBranchInternal"]);
   query.addObserver("extensions.fireform.fireform-activeuser", prefChangeObserver, false);
   query.addObserver("extensions.fireform.fireform-userList.", prefChangeObserver, false);
      
   /**
    * load form data
    */
   this.loadForm = function() {
      alert("Load Form");
   };
   
   /**
    * save form data
    */
   this.saveForm = function() {
      alert("Save Form");
   };
 
   /**
    * reset form data
    */
   this.resetForm = function() {
      alert("Reset Form");
   };

   /**
    * delete form data
    */
   this.deleteForm = function() {
      alert("Delete Form");
   };

   /**
    * open preference dialog
    */
   this.openPreferences = function() {
      window.open('chrome://fireform/content/xul/Preferences.xul', 'pref', 'chrome,dialog,centerscreen,modal');
   };

   /**
    * open help (webpage)
    */
   this.openHelp = function() {
      alert("Open Help");
   };
 
   /**
    * create the user menu
    */
   this.createUserMenu = function() {
      this.preferences.getPreference("fireform-userList").createUserMenu("fireform-button-user-select-pop");
   }
   
   /**
    * set a new active user from the
    * toolbar menupopup
    *
    * @param the menuitem triggered the change
    */
   this.changeUser = function(menuitem) {
      alert(menuitem);
   }
   
   /**
    * Object info as String
    *
    * @return string representation
    */
   this.toString = function() {
      return "Fireform Toolbar";
   };
}
